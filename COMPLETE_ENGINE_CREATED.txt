STATE MACHINE:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SEATING → PREP → PREFLOP → FLOP → TURN → RIVER → SHOWDOWN → CLEANUP

startHand():
  1. Move button clockwise
  2. Post blinds (SB, BB)
  3. Shuffle and deal hole cards
  4. Initialize PREFLOP betting

Each Street:
  1. Burn card (if postflop)
  2. Deal board cards (3/1/1)
  3. Reset contrib and currentBet
  4. Run betting round
  5. Check if only one player left
  6. Advance to next street or showdown


CORE DATA STRUCTURES:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Player {
  seatId: number
  stack: number              // Chips behind
  inHand: boolean            // Still in hand?
  folded: boolean            // Has folded?
  allIn: boolean             // Is all-in?
  hole: Card[]               // 2 hole cards
  contribStreet: number      // Contributed this street
  contribTotal: number       // Total contributed this hand
  isDealer: boolean
  isSmallBlind: boolean
  isBigBlind: boolean
}

Pot {
  cap: number                // Contribution cap for this pot
  amount: number             // Total chips in pot
  eligibleSeats: Set<number> // Players who can win
}

BettingRound {
  street: Street             // Current street
  currentBet: number         // Highest contrib this street
  lastRaiseSize: number      // Size of last raise
  actionCursor: number       // Current player to act
  lastAggressor: number      // Last bettor/raiser
  actionsThisRound: number   // Tracks if everyone acted
}

TableState {
  players: Player[]
  buttonIndex: number
  smallBlind: number
  bigBlind: number
  ante: number
  deck: Card[]               // Remaining cards
  burn: Card[]               // Burned cards
  board: Card[]              // Community cards
  pots: Pot[]                // Main + side pots
  round: BettingRound
  handNumber: number
  street: Street
}


BETTING LOGIC:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Key Calculations:
  toCall = max(0, currentBet - contribStreet)
  canCheck = toCall == 0 && !allIn
  minBet = bigBlind (when no bet yet)
  minRaiseTo = currentBet + lastRaiseSize (when facing bet)
  allInCap = contribTotal + stack

Action Handling:
  FOLD → player.folded = true, remove from all pots
  CHECK → no chips move, next player
  CALL → add toCall chips, match currentBet
  BET/RAISE → add chips, update currentBet and lastRaiseSize
  ALL-IN → add all chips, set allIn flag, create side pot

Bet/Raise Validation:
  • Opening bet: >= minBet (bigBlind)
  • Raise: new total >= currentBet + lastRaiseSize
  • Short all-in: allowed but doesn't reopen action
  • Can't bet more than stack

Betting Round Complete When:
  • Only 1 player left (others folded), OR
  • All players matched currentBet or are all-in
  • Everyone has had a chance to act


SIDE POT SYSTEM:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

When Player Goes All-In:
  1. Cap all existing pots at player's contribTotal
  2. Split overflow into new side pot
  3. Update eligibility:
     - Player eligible for pots up to their cap
     - Others eligible for higher pots

Example with 3 Players:
  Player A: 1000 chips all-in
  Player B: 2000 chips all-in
  Player C: 5000 chips active

  Pots Created:
  Main Pot (cap 1000):  3000 chips → A, B, C eligible
  Side Pot 1 (cap 2000): 2000 chips → B, C eligible
  Side Pot 2 (cap inf):  remaining → C eligible

Showdown:
  • Resolve main pot first (all 3 compete)
  • Then side pot 1 (B vs C)
  • Then side pot 2 (C wins by default)

EXAMPLE USAGE:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import { PokerEngine, Player } from './utils/PokerEngine';

// Create players
const players: Player[] = [
  {
    seatId: 0,
    stack: 10000,
    inHand: true,
    folded: false,
    allIn: false,
    hole: [],
    contribStreet: 0,
    contribTotal: 0,
    isMe: true,
    name: 'You',
    avatar: '🎮'
  },
  {
    seatId: 1,
    stack: 10000,
    inHand: true,
    folded: false,
    allIn: false,
    hole: [],
    contribStreet: 0,
    contribTotal: 0,
    name: 'AI Opponent',
    avatar: '🤖'
  }
];

// Create engine
const engine = new PokerEngine(
  players,
  0,              // Button at seat 0
  50,             // Small blind
  100,            // Big blind
  (state) => {
    // Update UI with new state
    console.log('State updated:', state);
  }
);

// Start hand
engine.startHand();

// Process action
engine.processAction({
  type: 'call',
  playerId: 0
});

// Check legal actions
const actions = engine.getLegalActions(0);
console.log('Legal actions:', actions);
