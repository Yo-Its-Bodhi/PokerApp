HOW THE BETTING SYSTEM WORKS NOW:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Game State Tracking:
  myBet       = How much YOU have bet this street
  opponentBet = How much OPPONENT has bet this street
  currentBet  = The amount you need to match (opponentBet - myBet)

Example Hand Flow:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PREFLOP:
  Blinds Posted:
    myBet: 500 (small blind)
    opponentBet: 1000 (big blind)
    currentBet: 500 (you need to call 500)
    
  Buttons: [FOLD] [CALL 500] [RAISE] [ALL IN]
  
  You Call 500:
    myBet: 1000
    opponentBet: 1000
    currentBet: 0 (bets are even)
    
  AI Raises to 3000:
    myBet: 1000
    opponentBet: 3000
    currentBet: 2000 (you need to call 2000)
    
  Buttons: [FOLD] [CALL 2000] [RAISE] [ALL IN]
  
  You Call 2000:
    myBet: 3000
    opponentBet: 3000
    currentBet: 0
    → Advance to FLOP

FLOP:
  Bets Reset:
    myBet: 0
    opponentBet: 0
    currentBet: 0
    
  Buttons: [FOLD] [CHECK] [BET] [ALL IN]
  
  You Bet 5000:
    myBet: 5000
    opponentBet: 0
    currentBet: 0 (waiting for opponent)
    
  AI Calls 5000:
    myBet: 5000
    opponentBet: 5000
    currentBet: 0
    → Advance to TURN
